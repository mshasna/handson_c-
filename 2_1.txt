using System;

namespace Practice2
{
    class Concat
    {
        static void Main(string[] args)
        {
            string name1, name2, name3;
            Console.WriteLine("Enter First Name : ");
            name1 = Console.ReadLine();
            Console.WriteLine("Enter Next Name : ");
            name2 = Console.ReadLine();
            name3 = name1 + " " + name2;
            Console.WriteLine(name3);
        }
    }
}
